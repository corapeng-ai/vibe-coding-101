phase('Intake') const profile = await agent( intakePrompt(args.query), { schema: PROFILE_SCHEMA } ) phase('Research') const [market, stocks, risk, news] = await parallel([ () => agent(marketPrompt(profile), { label: 'market', schema: MARKET_SCHEMA }), () => agent(screenerPrompt(profile), { label: 'screener', schema: STOCKS_SCHEMA }), () => agent(riskPrompt(profile), { label: 'risk', schema: RISK_SCHEMA }), () => agent(newsPrompt(profile), { label: 'news', schema: NEWS_SCHEMA }), ])
Build + Verify Loop
phase('Build Portfolio') let portfolio = await agent( buildPrompt(profile, market, stocks, risk, news), { schema: PORTFOLIO_SCHEMA } ) phase('Verify') let round = 0 while (round < 2) { const verdicts = await parallel([ () => agent(skepticPrompt( 'concentration', portfolio)), () => agent(skepticPrompt( 'bias', portfolio)), () => agent(skepticPrompt( 'freshness', portfolio)), ]) const issues = verdicts .filter(v => v?.hasIssue) if (!issues.length) break portfolio = await agent( revisePrompt(portfolio, issues)) round++ }
